pipeline {
  agent any
  stages {
    stage("Build"){
	steps{
	  echo 'Its Build Stage'
	}
    }
    stage("Test"){
	steps{
	  echo 'Its Test Stage'
	}
    }
    stage("Branch"){
        when {
	  not{
	     branch "master"
          }		
	}
	steps{
	  echo "Not Master Branch..!!"
	}
    }
    stage("Deploy"){
	steps{
	  echo 'Its DeployStage'
	}
    }
  }
}
