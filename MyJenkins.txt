pipeline {
  agent any
  stages {
    stage("Build"){
	steps{
	  echo 'Its Build Stage'
	}
    }
    stage("Test"){
	steps{
	  echo 'Its Test Stage'
	}
    }
    stage("Branch"){
        when {
	  not{
	     branch "master"
          }		
	}
	steps{
	  echo "Master Branch..!!"
	}
    }
    stage("Deploy"){
	parallel{
          stage("UAT Deploy"){
	    steps{
	      echo 'Its UAT Deploy Stage'
	    }
	  }
	  stage("Production Deploy"){
	    steps{
	      echo 'Its Production Deploy Stage'
	    }
	  }
	}
    }  
}
